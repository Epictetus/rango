public chainable(method = nil, &block)

Allows the definition of methods on a class that will be available via super.

Examples

    class Foo
      extend Rango::Chainable
      chainable do
        def hello
          "hello"
        end
      end
    end

    class Foo
      def hello
        super + " Merb!"
      end
    end

# Example with mixin: module TestMixin 

  extend Rango::Chainable
  chainable do
    def test
      "from mixin!"
    end
  end

end

class Test 

  include TestMixin
  def test
    "hello " + super
  end

end

puts Test.new.test

Foo.new.hello #=> “hello Merb!“
